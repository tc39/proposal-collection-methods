function merge(...iterables) {
    assert(isObject(this), 'this is not an object');
    assert(this instanceof Map, 'this is not a Map instance');
    for(const iterable of iterables) {
        for(const [key, value] of iterable) {
            this.set(key, value);
        }
    }

    return this;
}
